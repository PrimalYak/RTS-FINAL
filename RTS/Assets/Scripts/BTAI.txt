tree("Root")
	repeat
		parallel
			mute tree("SpawnTroops")
			mute tree("ManageTroops")
			mute tree("EnemiesInBase")
			
			

tree("ManageTroops")
		while not areEnemiesNearBase()
			tree("ManageActiveTroops")

tree("EnemiesInBase")
	while 
		areEnemiesNearBase()
		callDefendBase()

tree("ManageActiveTroops")
	sequence initializeMAT()
		while stillTroopsToManage()
			sequence
				manageTroop()
					
					
		
		
tree("SpawnTroops")
	
	fallback
		tree("PurchaseCounter")
		tree("GatherersAtMax")
		tree("PurchaseRandom")

		
tree("GatherersAtMax")
	sequence
		not gatherersAtMax()
		purchaseGatherer()
			

tree("PurchaseRandom")
	random
			purchaseArcher()
			purchaseWarrior()
			purchaseMage()

tree("PurchaseCounter")
	sequence
		enemyTroopsAlive()
		purchaseCounterTroop()
